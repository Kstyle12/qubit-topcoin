async function fetchNode(){

  try{

    const s=await fetch('http://localhost:5003/status');

    if(!s.ok)return;

    const d=await s.json();

    if(d.chain_length!==undefined)document.getElementById('ex-height').textContent=d.chain_length.toLocaleString();

    if(d.difficulty!==undefined)document.getElementById('ex-diff').textContent=d.difficulty;

    if(d.peers!==undefined)document.getElementById('ex-peers').textContent=d.peers;

    const c=await fetch('http://localhost:5003/chain');

    if(!c.ok)return;

    const cd=await c.json();

    const chain=cd.chain||[];

    const recent=chain.slice(-5).reverse();

    if(recent.length){

      document.getElementById('ex-blocks').innerHTML=recent.map(b=>`<div class="blk"><div><div class="blk-n">#${b.index}</div><div class="blk-m">${b.transactions?b.transactions.length:0} transactions</div></div><div class="blk-r">+50 QTOP</div></div>`).join('');

    }

  }catch(e){}

}

async function lookup(){

  const addr=document.getElementById('addr-in').value.trim();

  if(!addr)return;

  const el=document.getElementById('addr-out');

  el.textContent='Looking up...';

  try{

    const r=await fetch(`http://localhost:5003/balance/${addr}`);

    if(!r.ok){el.textContent='Could not reach node. Start a local node on port 5003.';return;}

    const d=await r.json();

    const q=(d.balance_cori/100000000).toFixed(8);

    el.innerHTML=`<div class="addr-row"><span class="addr-l">Address</span><span style="font-family:monospace;font-size:12px">${addr.slice(0,16)}...</span></div><div class="addr-row"><span class="addr-l">Balance</span><span style="color:#1D9E75;font-weight:500">${q} QTOP</span></div><div class="addr-row"><span class="addr-l">In cori</span><span>${(d.balance_cori||0).toLocaleString()}</span></div>`;

  }catch(e){el.textContent='Could not reach node. Make sure your local node is running on port 5003.';}

}
